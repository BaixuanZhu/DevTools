**Goal:** 构建一个在线 JSON 对比工具，支持语义模式和严格模式，通过并排视图可视化展示差异。

**Architecture:** 纯函数模块 `json-diff.ts` 负责所有 diff 算法（语义递归对比 + LCS 行级对比），不依赖 Vue。Vue 组件 `JsonDiff.vue` 负责输入、渲染、交互，调用纯函数获取结果。大文件通过 Web Worker 异步处理。

**Tech Stack:** Vue 3 Composition API + TypeScript + Vitest + @headlessui/vue + Tailwind CSS v4

```ts
export interface LineDiff {

  rightLineNo?: number;
```

```ts
// ---- 嵌套深度检测 ----

/**
 * 扫描 JSON 文本的最大嵌套深度。
 *
 * 逐字符遍历，遇 `{` 或 `[` 深度 +1，遇 `}` 或 `]` 深度 -1。
 * 正确处理字符串内的括号（忽略字符串和转义字符内的结构字符）。
 * 时间复杂度 O(n)。
 *
 * @param jsonText - 原始 JSON 文本
 * @returns 最大嵌套深度（0 表示无对象/数组）
 */
export function measureMaxDepth(jsonText: string): number {
  let maxDepth = 0;
  let currentDepth = 0;
  let inString = false;
  let i = 0;

  while (i < jsonText.length) {
    const ch = jsonText[i];

    if (inString) {
      if (ch === '\\') {
        // 跳过转义字符
        i += 2;
        continue;
      }
      if (ch === '"') {
        inString = false;
      }
    } else {
      if (ch === '"') {
        inString = true;
      } else if (ch === '{' || ch === '[') {
        currentDepth++;
        if (currentDepth > maxDepth) {
          maxDepth = currentDepth;
        }
      } else if (ch === '}' || ch === ']') {
        currentDepth--;
      }
    }
    i++;
  }

  return maxDepth;
}
```

```ts
// ---- JSON 解析辅助 ----

/**
 * 安全解析 JSON 文本，带深度拦截和错误信息提取。
 *
 * 解析前先通过 measureMaxDepth 检查嵌套深度，超过 256 层直接拒绝。
 * 解析后检查节点总数，超过 500,000 也拒绝。
 *
 * @param jsonText - 原始 JSON 文本
 * @returns 解析结果（成功时含数据和节点数，失败时含错误信息）
 */
export function parseJsonSafe(jsonText: string): ParseOutcome {
  const depth = measureMaxDepth(jsonText);
  if (depth > MAX_DEPTH_LIMIT) {
    return {
      ok: false,
      error: `JSON 嵌套层级过深（${depth} 层），最大支持 ${MAX_DEPTH_LIMIT} 层`,
    };
  }

  try {
    const data = JSON.parse(jsonText);
    const nodeCount = countNodes(data);
    if (nodeCount > MAX_NODE_COUNT) {
      return {
        ok: false,
        error: `节点数量过多（${nodeCount.toLocaleString()}），最大支持 ${MAX_NODE_COUNT.toLocaleString()}`,
      };
    }
    return { ok: true, data, nodeCount };
  } catch (e) {
    const msg = e instanceof Error ? e.message : String(e);
    const pos = parseErrorPosition(msg, jsonText);
    let error = `JSON 语法错误：${translateErrorMessage(msg)}`;
    if (pos.line !== undefined && pos.column !== undefined) {
      error += `（第 ${pos.line} 行，第 ${pos.column} 列）`;
    }
    return { ok: false, error, ...pos };
  }
}

/**
 * 递归计算 JSON 对象的节点总数。
 *
 * @param obj - 已解析的 JSON 值
 * @returns 节点数量
 */
export function countNodes(obj: unknown): number {
  if (obj === null || typeof obj !== 'object') return 1;
  if (Array.isArray(obj)) {
    return obj.reduce((sum, item) => sum + countNodes(item), 0);
  }
  return Object.values(obj as Record<string, unknown>).reduce<number>(
    (sum, val) => sum + countNodes(val),
    0,
  );
}

/**
 * 检查输入文本大小是否超限。
 *
 * @param input - 原始输入字符串
 * @returns 'ok' 或 'error'
 */
export function checkInputSize(input: string): 'ok' | 'error' {
  const size = new TextEncoder().encode(input).length;
  return size > INPUT_SIZE_LIMIT ? 'error' : 'ok';
}

// ---- 内部辅助 ----

/**
 * 从 JSON.parse 错误信息中提取行列号。
 */
function parseErrorPosition(msg: string, input: string): { line?: number; column?: number } {
  const posMatch = msg.match(/position\s+(\d+)/i);
  if (!posMatch) return {};
  const pos = parseInt(posMatch[1], 10);
  const beforeError = input.substring(0, pos);
  const line = beforeError.split('\n').length;
  const lastNewline = beforeError.lastIndexOf('\n');
  const column = lastNewline === -1 ? pos + 1 : pos - lastNewline;
  return { line, column };
}

/**
 * 翻译英文错误消息为中文。
 */
function translateErrorMessage(msg: string): string {
  return msg
    .replace(/Unexpected token/i, '意外的字符')
    .replace(/Unexpected end of JSON input/i, 'JSON 数据意外结束')
    .replace(/Expected property name/i, '期望属性名')
    .replace(/Expected ':'/i, '期望冒号 ":"')
    .replace(/Bad control character/i, '非法控制字符')
    .replace(/Bad string/i, '非法字符串')
    .replace(/Bad escape sequence/i, '非法转义序列');
}
```

```ts
/** 语义 diff 内部使用的递归上下文 */
interface DiffContext {
  ignoreArrayOrder: boolean;
  depth: number;
  truncated: boolean;
}

/**
 * 语义 diff：递归对比两份已解析的 JSON 对象。
 *
 * 对比前先做快速短路（排序后序列化相等则直接返回无差异）。
 * 递归深度超过 MAX_DIFF_DEPTH 时截断，当前节点标记为 modified。
 *
 * @param left - 左侧已解析值
 * @param right - 右侧已解析值
 * @param options - 对比选项
 * @returns 语义对比结果
 */
export function semanticDiff(
  left: unknown,
  right: unknown,
  options: { ignoreArrayOrder?: boolean } = {},
): DiffResult {
  const ctx: DiffContext = {
    ignoreArrayOrder: options.ignoreArrayOrder ?? false,
    depth: 0,
    truncated: false,
  };

  // 快速短路：排序后序列化相同则无差异
  if (isPrimitive(left) || isPrimitive(right)) {
    // 基本类型直接比较
  } else {
    try {
      const leftStr = stableStringify(left);
      const rightStr = stableStringify(right);
      if (leftStr === rightStr) {
        return { items: [], summary: { added: 0, removed: 0, modified: 0, unchanged: 0 } };
      }
    } catch {
      // 序列化失败，继续正常对比
    }
  }

  const items: DiffItem[] = [];
  const summary = { added: 0, removed: 0, modified: 0, unchanged: 0 };

  compareValues(left, right, '$', ctx, items, summary);

  if (ctx.truncated) {
    // 调用方可通过 items 中是否存在截断标记来判断
  }

  return { items, summary };
}

/**
 * 递归对比两个值。
 */
function compareValues(
  left: unknown,
  right: unknown,
  path: string,
  ctx: DiffContext,
  items: DiffItem[],
  summary: { added: number; removed: number; modified: number; unchanged: number },
): void {
  // 深度截断
  if (ctx.depth > MAX_DIFF_DEPTH) {
    ctx.truncated = true;
    items.push({ type: 'modified', path, oldValue: left, newValue: right });
    summary.modified++;
    return;
  }

  // null 处理
  if (left === null && right === null) {
    return; // 相同
  }
  if (left === null || right === null) {
    if (left === right) return;
    items.push({ type: 'modified', path, oldValue: left, newValue: right });
    summary.modified++;
    return;
  }

  // 基本类型
  if (isPrimitive(left) && isPrimitive(right)) {
    if (left === right) return;
    // -0 === 0 为 true，不做特殊处理
    items.push({ type: 'modified', path, oldValue: left, newValue: right });
    summary.modified++;
    return;
  }

  // 类型不同
  const leftType = typeof left === 'object' ? (Array.isArray(left) ? 'array' : 'object') : typeof left;
  const rightType = typeof right === 'object' ? (Array.isArray(right) ? 'array' : 'object') : typeof right;

  if (leftType !== rightType) {
    items.push({ type: 'modified', path, oldValue: left, newValue: right });
    summary.modified++;
    return;
  }

  // 对象对比
  if (leftType === 'object' && !Array.isArray(left) && !Array.isArray(right)) {
    compareObjects(
      left as Record<string, unknown>,
      right as Record<string, unknown>,
      path,
      ctx,
      items,
      summary,
    );
    return;
  }

  // 数组对比
  if (leftType === 'array' && Array.isArray(left) && Array.isArray(right)) {
    if (ctx.ignoreArrayOrder) {
      compareArraysUnordered(left, right, path, ctx, items, summary);
    } else {
      compareArraysOrdered(left, right, path, ctx, items, summary);
    }
  }
}

/**
 * 对象键值对比。
 */
function compareObjects(
  left: Record<string, unknown>,
  right: Record<string, unknown>,
  path: string,
  ctx: DiffContext,
  items: DiffItem[],
  summary: { added: number; removed: number; modified: number; unchanged: number },
): void {
  const allKeys = new Set([...Object.keys(left), ...Object.keys(right)]);

  for (const key of allKeys) {
    const escapedKey = key.includes('.') || key.includes('[') || key.includes(']') ? `["${key}"]` : `.${key}`;
    const childPath = path === '$' ? `$${escapedKey}` : `${path}${escapedKey}`;

    const inLeft = key in left;
    const inRight = key in right;

    if (inLeft && !inRight) {
      items.push({ type: 'removed', path: childPath, oldValue: left[key] });
      summary.removed++;
    } else if (!inLeft && inRight) {
      items.push({ type: 'added', path: childPath, newValue: right[key] });
      summary.added++;
    } else {
      ctx.depth++;
      compareValues(left[key], right[key], childPath, ctx, items, summary);
      ctx.depth--;
    }
  }
}

/**
 * 数组按索引顺序对比。
 */
function compareArraysOrdered(
  left: unknown[],
  right: unknown[],
  path: string,
  ctx: DiffContext,
  items: DiffItem[],
  summary: { added: number; removed: number; modified: number; unchanged: number },
): void {
  const maxLen = Math.max(left.length, right.length);

  for (let i = 0; i < maxLen; i++) {
    const childPath = `${path}[${i}]`;

    if (i < left.length && i < right.length) {
      ctx.depth++;
      compareValues(left[i], right[i], childPath, ctx, items, summary);
      ctx.depth--;
    } else if (i < left.length) {
      items.push({ type: 'removed', path: childPath, oldValue: left[i] });
      summary.removed++;
    } else {
      items.push({ type: 'added', path: childPath, newValue: right[i] });
      summary.added++;
    }
  }
}

/**
 * 数组无序对比（多重集匹配）。
 */
function compareArraysUnordered(
  left: unknown[],
  right: unknown[],
  path: string,
  ctx: DiffContext,
  items: DiffItem[],
  summary: { added: number; removed: number; modified: number; unchanged: number },
): void {
  // 将元素序列化为稳定字符串，用 Map 计数
  const leftCounts = new Map<string, { value: unknown; count: number }>();
  const rightCounts = new Map<string, { value: unknown; count: number }>();

  for (const item of left) {
    const key = stableStringify(item);
    const entry = leftCounts.get(key);
    if (entry) {
      entry.count++;
    } else {
      leftCounts.set(key, { value: item, count: 1 });
    }
  }

  for (const item of right) {
    const key = stableStringify(item);
    const entry = rightCounts.get(key);
    if (entry) {
      entry.count++;
    } else {
      rightCounts.set(key, { value: item, count: 1 });
    }
  }

  // 比较计数差异
  const allKeys = new Set([...leftCounts.keys(), ...rightCounts.keys()]);

  for (const key of allKeys) {
    const leftEntry = leftCounts.get(key);
    const rightEntry = rightCounts.get(key);
    const leftCount = leftEntry?.count ?? 0;
    const rightCount = rightEntry?.count ?? 0;

    if (leftCount === rightCount) continue; // 数量相同

    const diff = Math.abs(rightCount - leftCount);
    if (rightCount > leftCount) {
      for (let i = 0; i < diff; i++) {
        items.push({ type: 'added', path: `${path}[?]`, newValue: rightEntry!.value });
        summary.added++;
      }
    } else {
      for (let i = 0; i < diff; i++) {
        items.push({ type: 'removed', path: `${path}[?]`, oldValue: leftEntry!.value });
        summary.removed++;
      }
    }
  }
}
```

```ts
// ---- 严格文本 Diff (LCS) ----

/**
 * 严格文本 diff：格式化后按行 LCS 对比。
 *
 * 将两份 JSON 格式化为标准缩进文本，拆行后使用 LCS 算法求行级对齐。
 * 行数超过 LCS_GREEDY_THRESHOLD 时切换为贪心 diff（O(N) 空间）。
 *
 * @param leftJson - 左侧 JSON 文本
 * @param rightJson - 右侧 JSON 文本
 * @param indentSize - 格式化缩进空格数
 * @returns 行级差异结果
 */
export function strictDiff(leftJson: string, rightJson: string, indentSize: 2 | 4 = 2): StrictDiffResult {
  // 格式化
  let leftFormatted: string;
  let rightFormatted: string;
  try {
    leftFormatted = JSON.stringify(JSON.parse(leftJson), null, indentSize);
  } catch {
    leftFormatted = leftJson;
  }
  try {
    rightFormatted = JSON.stringify(JSON.parse(rightJson), null, indentSize);
  } catch {
    rightFormatted = rightJson;
  }

  const leftLines = leftFormatted.split('\n');
  const rightLines = rightFormatted.split('\n');

  // 快速短路
  if (leftFormatted === rightFormatted) {
    return {
      lines: leftLines.map((content, i) => ({
        type: 'unchanged' as const,
        leftLineNo: i + 1,
        rightLineNo: i + 1,
        content,
      })),
      summary: { added: 0, removed: 0, unchanged: leftLines.length },
    };
  }

  // 选择算法
  const totalLines = leftLines.length + rightLines.length;
  const lines: LineDiff[] = [];

  if (totalLines > LCS_GREEDY_THRESHOLD) {
    greedyDiff(leftLines, rightLines, lines);
  } else {
    lcsDiff(leftLines, rightLines, lines);
  }

  // 计算摘要
  const summary = {
    added: lines.filter(l => l.type === 'added').length,
    removed: lines.filter(l => l.type === 'removed').length,
    unchanged: lines.filter(l => l.type === 'unchanged').length,
  };

  return { lines, summary };
}

/**
 * 标准 LCS diff（O(N×M) 时间和空间）。
 */
function lcsDiff(oldLines: string[], newLines: string[], result: LineDiff[]): void {
  const M = oldLines.length;
  const N = newLines.length;

  // 构建 DP 表
  const dp: number[][] = Array.from({ length: M + 1 }, () => new Array(N + 1).fill(0));

  for (let i = 1; i <= M; i++) {
    for (let j = 1; j <= N; j++) {
      if (oldLines[i - 1] === newLines[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }

  // 回溯
  const rawLines: LineDiff[] = [];
  let i = M;
  let j = N;

  while (i > 0 || j > 0) {
    if (i > 0 && j > 0 && oldLines[i - 1] === newLines[j - 1]) {
      rawLines.push({ type: 'unchanged', leftLineNo: i, rightLineNo: j, content: oldLines[i - 1] });
      i--;
      j--;
    } else if (j > 0 && (i === 0 || dp[i][j - 1] >= dp[i - 1][j])) {
      rawLines.push({ type: 'added', rightLineNo: j, content: newLines[j - 1] });
      j--;
    } else {
      rawLines.push({ type: 'removed', leftLineNo: i, content: oldLines[i - 1] });
      i--;
    }
  }

  rawLines.reverse();

  // 填充对侧行号（保持行号连续性）
  assignLineNumbers(rawLines, result);
}

/**
 * 贪心 diff（O(N) 空间）：使用简单的行匹配策略。
 */
function greedyDiff(oldLines: string[], newLines: string[], result: LineDiff[]): void {
  const rawLines: LineDiff[] = [];
  let oi = 0;
  let ni = 0;

  // 构建行到索引的映射（用于快速查找）
  const newLineMap = new Map<string, number[]>();
  for (let k = 0; k < newLines.length; k++) {
    const line = newLines[k];
    const indices = newLineMap.get(line);
    if (indices) {
      indices.push(k);
    } else {
      newLineMap.set(line, [k]);
    }
  }

  const usedNew = new Set<number>();

  while (oi < oldLines.length || ni < newLines.length) {
    if (oi < oldLines.length) {
      const oldLine = oldLines[oi];
      const candidates = newLineMap.get(oldLine);
      let matchedIdx = -1;

      if (candidates) {
        for (const idx of candidates) {
          if (idx >= ni && !usedNew.has(idx)) {
            matchedIdx = idx;
            break;
          }
        }
      }

      if (matchedIdx >= 0) {
        // 输出 ni 到 matchedIdx 之间的新增行
        while (ni < matchedIdx) {
          rawLines.push({ type: 'added', rightLineNo: ni + 1, content: newLines[ni] });
          ni++;
        }
        rawLines.push({ type: 'unchanged', leftLineNo: oi + 1, rightLineNo: ni + 1, content: oldLine });
        usedNew.add(ni);
        oi++;
        ni++;
      } else {
        // 在 ni 之前查找是否有新增行要优先输出
        rawLines.push({ type: 'removed', leftLineNo: oi + 1, content: oldLine });
        oi++;
      }
    } else {
      rawLines.push({ type: 'added', rightLineNo: ni + 1, content: newLines[ni] });
      ni++;
    }
  }

  assignLineNumbers(rawLines, result);
}

/**
 * 为 diff 行分配对侧行号（确保 removed/added 行的对侧有占位行号）。
 */
function assignLineNumbers(rawLines: LineDiff[], result: LineDiff[]): void {
  let leftNo = 0;
  let rightNo = 0;

  for (const line of rawLines) {
    if (line.type === 'unchanged') {
      leftNo = line.leftLineNo!;
      rightNo = line.rightLineNo!;
      result.push(line);
    } else if (line.type === 'removed') {
      leftNo = line.leftLineNo!;
      result.push({ ...line, leftLineNo });
    } else {
      // added
      rightNo = line.rightLineNo!;
      result.push({ ...line, rightLineNo });
    }
  }
}
```

```ts
// ---- Unified Diff 格式化 ----

/**
 * 将 LineDiff 数组格式化为标准 unified diff 文本。
 *
 * 输出格式：`---` / `+++` 头部 + `@@ ... @@` 范围头 + `-` / `+` 前缀行。
 *
 * @param lines - 行级差异数组
 * @returns unified diff 格式文本
 */
export function formatUnifiedDiff(lines: LineDiff[]): string {
  if (lines.length === 0) return '';

  const parts: string[] = ['--- left.json', '+++ right.json'];

  // 将行按 unchanged 分组，提取差异块
  let i = 0;
  while (i < lines.length) {
    // 找到下一个差异块
    while (i < lines.length && lines[i].type === 'unchanged') i++;
    if (i >= lines.length) break;

    // 收集差异块（差异行 + 上下文）
    const blockStart = Math.max(0, i - DEFAULT_CONTEXT_LINES);
    let blockEnd = i;
    while (blockEnd < lines.length && lines[blockEnd].type !== 'unchanged') blockEnd++;
    // 包含后续上下文
    const contextAfter = Math.min(DEFAULT_CONTEXT_LINES, lines.length - blockEnd);
    blockEnd += contextAfter;

    // 计算范围
    const oldStart = lines[blockStart]?.leftLineNo ?? 1;
    const newStart = lines[blockStart]?.rightLineNo ?? 1;
    let oldCount = 0;
    let newCount = 0;
    for (let j = blockStart; j < blockEnd; j++) {
      if (lines[j].type === 'removed' || lines[j].type === 'unchanged') oldCount++;
      if (lines[j].type === 'added' || lines[j].type === 'unchanged') newCount++;
    }

    parts.push(`@@ -${oldStart},${oldCount} +${newStart},${newCount} @@`);

    for (let j = blockStart; j < blockEnd; j++) {
      const line = lines[j];
      if (line.type === 'unchanged') {
        parts.push(` ${line.content}`);
      } else if (line.type === 'removed') {
        parts.push(`-${line.content}`);
      } else {
        parts.push(`+${line.content}`);
      }
    }

    i = blockEnd;
  }

  return parts.join('\n');
}
```

```ts
    let result: WorkerSuccessResponse['result'];

    if (mode === 'semantic') {
      const diffResult = semanticDiff(leftParsed.data, rightParsed.data, {
        ignoreArrayOrder: options.ignoreArrayOrder,
      });
      result = diffResult;
    } else {
      const diffResult = strictDiff(leftJson, rightJson, options.indentSize);
      result = diffResult;
    }

    const resp: WorkerResponse = { ok: true, result };
    self.postMessage(resp);
  } catch (err) {
    const msg = err instanceof Error ? err.message : String(err);
    const resp: WorkerResponse = { ok: false, error: `对比过程中发生错误：${msg}` };
    self.postMessage(resp);
  }
};
```

```ts
/** 执行对比 */
function handleCompare(): void {
  leftError.value = '';
  rightError.value = '';
  runtimeError.value = '';
  depthWarning.value = '';
  isIdentical.value = false;
  isExpanded.value = false;

  // 校验空输入
  const leftEmpty = !leftInput.value.trim();
  const rightEmpty = !rightInput.value.trim();

  if (leftEmpty && rightEmpty) {
    leftError.value = '请输入需要对比的 JSON 内容';
    return;
  }
  if (leftEmpty) {
    leftError.value = '请输入 JSON 内容';
    return;
  }
  if (rightEmpty) {
    rightError.value = '请输入 JSON 内容';
    return;
  }

  // 校验大小
  if (checkInputSize(leftInput.value) === 'error') {
    const size = (new TextEncoder().encode(leftInput.value).length / (1024 * 1024)).toFixed(1);
    leftError.value = `内容过大（${size}MB），最大支持 10MB`;
    return;
  }
  if (checkInputSize(rightInput.value) === 'error') {
    const size = (new TextEncoder().encode(rightInput.value).length / (1024 * 1024)).toFixed(1);
    rightError.value = `内容过大（${size}MB），最大支持 10MB`;
    return;
  }

  // 深度检查
  const leftDepth = measureMaxDepth(leftInput.value);
  const rightDepth = measureMaxDepth(rightInput.value);
  if (leftDepth > MAX_DEPTH_LIMIT) {
    leftError.value = `JSON 嵌套层级过深（${leftDepth} 层），最大支持 ${MAX_DEPTH_LIMIT} 层`;
    return;
  }
  if (rightDepth > MAX_DEPTH_LIMIT) {
    rightError.value = `JSON 嵌套层级过深（${rightDepth} 层），最大支持 ${MAX_DEPTH_LIMIT} 层`;
    return;
  }

  // 判断是否需要 Worker
  const leftSize = new TextEncoder().encode(leftInput.value).length;
  const rightSize = new TextEncoder().encode(rightInput.value).length;

  if (leftSize + rightSize > WORKER_THRESHOLD) {
    compareWithWorker();
  } else {
    compareSync();
  }
}

/** 同步对比 */
function compareSync(): void {
  try {
    const leftParsed = parseJsonSafe(leftInput.value);
    if (!leftParsed.ok) {
      leftError.value = leftParsed.error;
      return;
    }

    const rightParsed = parseJsonSafe(rightInput.value);
    if (!rightParsed.ok) {
      rightError.value = rightParsed.error;
      return;

    runtimeError.value = '对比过程中发生错误，请检查输入或减小数据量';
  }
}

    initWorker();

  worker!.postMessage({
    leftJson: leftInput.value,
    rightJson: rightInput.value,
    mode: diffMode.value,
    options: options.value,

}

/** 折叠 unchanged 行，保留差异上下文 */
function foldUnchanged(lines: LineDiff[], context: number): LineDiff[] {
  const result: LineDiff[] = [];
  // 找出所有差异行的索引
  const diffIndices: number[] = [];
  for (let i = 0; i < lines.length; i++) {
    if (lines[i].type !== 'unchanged') diffIndices.push(i);
  }

  if (diffIndices.length === 0) return lines;

  // 收集需要显示的索引范围
  const visibleSet = new Set<number>();
  for (const idx of diffIndices) {
    for (let j = Math.max(0, idx - context); j <= Math.min(lines.length - 1, idx + context); j++) {
      visibleSet.add(j);
    }
  }

  // 构建结果，在间隔处插入折叠标记
  let lastVisible = -1;
  for (const idx of [...visibleSet].sort((a, b) => a - b)) {
    if (lastVisible >= 0 && idx > lastVisible + 1) {
      // 插入折叠占位（用特殊标记）
      result.push({ type: 'unchanged', content: `@@ ${idx - lastVisible - 1} unchanged lines @@` });
    }
    result.push(lines[idx]);
    lastVisible = idx;
  }

  return result;
}

/** 折叠 unchanged 项（语义模式） */
function foldUnchangedItems(items: DiffItem[], context: number): DiffItem[] {
  // 语义模式不折叠（无行对齐概念），直接返回
  return items;
}

/** 将语义 diff 结果格式化为文本 */
function formatSemanticAsText(result: DiffResult): string {
  const parts: string[] = [];
  for (const item of result.items) {
    if (item.type === 'added') {
      parts.push(`+ ${item.path}: ${JSON.stringify(item.newValue)}`);
    } else if (item.type === 'removed') {
      parts.push(`- ${item.path}: ${JSON.stringify(item.oldValue)}`);
    } else if (item.type === 'modified') {
      parts.push(`~ ${item.path}: ${JSON.stringify(item.oldValue)} → ${JSON.stringify(item.newValue)}`);
    }
  }
  return parts.join('\n');
}

/** 获取差异类型对应的 CSS 类 */
function getDiffLineClass(type: 'added' | 'removed' | 'unchanged'): string {
  switch (type) {
    case 'added': return 'bg-green-50/50 text-success';
    case 'removed': return 'bg-red-50/50 text-error';
    default: return 'text-text';
  }
}

/** 获取差异类型前缀图标 */
function getDiffPrefix(type: 'added' | 'removed' | 'unchanged'): string {
  switch (type) {
    case 'added': return '+';
    case 'removed': return '-';
    default: return ' ';
  }
}
```

```html
              rightDragging ? 'border-accent ring-2 ring-accent/20' : 'border-border',
              rightError ? 'border-error' : '',
            ]"
            aria-label="修改后 JSON 输入"
            placeholder="粘贴或输入修改后的 JSON，支持拖拽 .json 文件…"
            spellcheck="false"
            @dragover.prevent="rightDragging = true"
            @dragleave="rightDragging = false"
            @drop="handleDrop('right', $event)"
          />
          <div v-if="rightError" class="mt-1 text-[0.75rem] text-error">{{ rightError }}</div>
          <div class="flex gap-2 mt-2">
            <button
              class="px-3 py-1 border border-border rounded-sm bg-card text-muted text-[0.75rem] font-sans cursor-pointer transition-[background-color,color] duration-150 hover:bg-hover hover:text-text"
              @click="handlePaste('right')"
            >
              📋 粘贴
            </button>
            <button
              class="px-3 py-1 border border-border rounded-sm bg-card text-muted text-[0.75rem] font-sans cursor-pointer transition-[background-color,color] duration-150 hover:bg-hover hover:text-text"
              @click="rightFileInput?.click()"
            >
              📁 上传文件
            </button>
            <input
              ref="rightFileInput"
              type="file"
              accept=".json,.txt,application/json,text/plain"
              class="hidden"
              @change="handleFileUpload('right', $event)"
            />
          </div>
        </div>
      </template>
    </ResponsiveWorkspace>

    <!-- 摘要面板 -->
    <div
      v-if="hasResult && summary"
      class="mt-6 px-4 py-3 border border-border rounded-sm bg-card"
      aria-live="polite"
    >
      <div v-if="isIdentical" class="text-sm text-success font-sans">
        ✅ 两份 JSON 完全一致，无差异
      </div>
      <div v-else class="flex flex-wrap gap-4 text-sm font-sans">
        <span class="text-text">✅ {{ summary.unchanged }} 处一致</span>
        <span class="text-success">🟢 {{ summary.added }} 处新增</span>
        <span class="text-error">🔴 {{ summary.removed }} 处删除</span>
        <span v-if="summary.modified !== undefined" class="text-accent">🟡 {{ summary.modified }} 处修改</span>
      </div>
      <div v-if="depthWarning" class="mt-1 text-[0.75rem] text-accent">{{ depthWarning }}</div>
    </div>

    <!-- 运行时错误 -->
    <div
      v-if="runtimeError"
      class="mt-4 px-4 py-3 border border-error/30 rounded-sm bg-error/5 text-sm text-error"
    >
      {{ runtimeError }}
    </div>

    <!-- Diff 视图 -->
    <div v-if="hasResult && !isIdentical" class="mt-4">
      <!-- 严格模式：并排 diff -->
      <template v-if="diffMode === 'strict'">
        <div class="grid grid-cols-1 lg:grid-cols-2 gap-0 border border-border rounded-sm overflow-hidden">
          <!-- 左侧面板 -->
          <div
            ref="leftPanelRef"
            class="overflow-auto bg-card max-h-[50vh]"
            @scroll="handleScroll('left')"
          >
            <table class="w-full border-collapse font-mono text-sm leading-6">
              <tbody>
                <tr
                  v-for="(line, idx) in foldedLines"
                  :key="'l' + idx"
                  :class="getDiffLineClass(line.type)"
                >
                  <td class="px-2 py-0 text-right text-muted select-none w-8 border-r border-border/50">{{ line.leftLineNo ?? '' }}</td>
                  <td class="px-1 py-0 w-4 select-none">{{ getDiffPrefix(line.type === 'added' ? 'added' : line.type) }}</td>
                  <td class="px-2 py-0 whitespace-pre">{{ line.content }}</td>
                </tr>
              </tbody>
            </table>
          </div>
          <!-- 右侧面板 -->
          <div
            ref="rightPanelRef"
            class="overflow-auto bg-card max-h-[50vh] border-l border-border"
            @scroll="handleScroll('right')"
          >
            <table class="w-full border-collapse font-mono text-sm leading-6">
              <tbody>
                <tr
                  v-for="(line, idx) in foldedLines"
                  :key="'r' + idx"
                  :class="getDiffLineClass(line.type)"
                >
                  <td class="px-2 py-0 text-right text-muted select-none w-8 border-r border-border/50">{{ line.rightLineNo ?? '' }}</td>
                  <td class="px-1 py-0 w-4 select-none">{{ getDiffPrefix(line.type === 'removed' ? 'removed' : line.type) }}</td>
                  <td class="px-2 py-0 whitespace-pre">{{ line.content }}</td>
                </tr>
              </tbody>
            </table>
          </div>
        </div>
      </template>

      <!-- 语义模式：扁平化键值列表 -->
      <template v-else>
        <div class="border border-border rounded-sm overflow-hidden">
          <div class="overflow-auto bg-card max-h-[50vh]">
            <table class="w-full border-collapse font-mono text-sm leading-6">
              <thead>
                <tr class="border-b border-border bg-hover/50">
                  <th class="px-3 py-1.5 text-left text-muted text-[0.75rem] font-sans w-12"></th>
                  <th class="px-3 py-1.5 text-left text-muted text-[0.75rem] font-sans">路径</th>
                  <th class="px-3 py-1.5 text-left text-muted text-[0.75rem] font-sans">旧值</th>
                  <th class="px-3 py-1.5 text-left text-muted text-[0.75rem] font-sans">新值</th>
                </tr>
              </thead>
              <tbody>
                <tr
                  v-for="(item, idx) in diffItems"
                  :key="idx"
                  :class="getDiffLineClass(item.type)"
                >
                  <td class="px-2 py-0 text-center select-none">{{ getDiffPrefix(item.type) }}</td>
                  <td class="px-3 py-0 whitespace-pre text-text">{{ item.path }}</td>
                  <td class="px-3 py-0 whitespace-pre max-w-xs truncate">{{ JSON.stringify(item.oldValue) }}</td>
                  <td class="px-3 py-0 whitespace-pre max-w-xs truncate">{{ JSON.stringify(item.newValue) }}</td>
                </tr>
              </tbody>
            </table>
          </div>
        </div>
      </template>

      <!-- 展开/折叠控制 -->
      <div v-if="needsAutoFold" class="mt-2 text-center">
        <button
          v-if="!isExpanded"
          class="px-4 py-1.5 border border-border rounded-sm bg-card text-muted text-[0.75rem] font-sans cursor-pointer transition-[background-color,color] duration-150 hover:bg-hover hover:text-text"
          @click="handleExpandAll"
        >
          展开全部（{{ displayCount }} 行）
        </button>
      </div>
    </div>

    <!-- 初始占位 -->
    <div
      v-if="!hasResult && !runtimeError"
      class="mt-6 p-8 border border-border rounded-sm bg-card text-center text-muted text-sm"
    >
      输入两份 JSON 后点击「对比」查看差异
    </div>

    <!-- 展开确认对话框 -->
    <div
      v-if="showExpandConfirm"
      class="fixed inset-0 z-50 flex items-center justify-center bg-black/30"
    >
      <div class="bg-card border border-border rounded-md p-6 max-w-sm mx-4 shadow-lg">
        <p class="text-sm text-text font-sans mb-4">
          展开后将显示 {{ pendingExpandLines }} 行，可能影响流畅度，是否继续？
        </p>
        <div class="flex justify-end gap-2">
          <button
            class="px-4 py-2 border border-border rounded-sm bg-card text-text text-sm font-sans cursor-pointer hover:bg-hover"
            @click="cancelExpand"
          >
            取消
          </button>
          <button
            class="px-4 py-2 rounded-sm bg-accent text-white text-sm font-sans cursor-pointer hover:bg-accent/90"
            @click="confirmExpand"
          >
            确认展开
          </button>
        </div>
      </div>
    </div>
  </div>
</template>
```

注意：`leftTextarea` ref 已在 Step 1 状态部分声明，此处 `<textarea>` 标签添加 `ref="leftTextarea"` 即可。

> **关于虚拟滚动**：计划中定义了 `needsVirtualScroll`、`virtualScrollTop`、`visibleRange` 等计算属性，但初始版本模板使用 `v-for` 直接遍历 `foldedLines`/`diffItems`。折叠机制（unchanged 上下文保留 3 行）已将 DOM 节点控制在合理范围。如果后续需要处理超大 diff（> 10000 差异行），可升级为基于 `visibleRange` 的虚拟滚动渲染。
